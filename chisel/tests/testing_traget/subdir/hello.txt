/**
 * Endpoint Test Script - Simulates browser behavior with cookie-based auth
 * Tests all dashboard-be endpoints for accessibility
 * 
 * Usage: 
 *   node script/test-endpoints.js       # Normal mode
 *   node script/test-endpoints.js -v    # Verbose mode (logs responses as CSV)
 */

const BASE_URL = process.env.API_URL || 'https://tbv-dashboard-be-dev-694910840590.us-east1.run.app';
const TEST_EMAIL = process.env.TEST_EMAIL || 'tester@test.com';
const TEST_PASSWORD = process.env.TEST_PASSWORD || '111111';

// Parse command line args
const VERBOSE = process.argv.includes('-v') || process.argv.includes('--verbose');

// Cookie jar to store tokens like a browser
let cookies = {
  accessToken: null,
  refreshToken: null
};

// Results tracking
const res = {
  passed: [],
  failed: [],
  skipped: []
};

/**
 * Parse Set-Cookie headers and store in cookie jar
 */
function parseCookies(response) {
  let setCookieHeaders = [];
  
  if (typeof response.headers.getSetCookie === 'function') {
    setCookieHeaders = response.headers.getSetCookie();
  } else {
    const raw = response.headers.get('set-cookie');
    if (raw) {
      setCookieHeaders = raw.split(/,(?=\s*(?:accessToken|refreshToken)=)/);
    }
  }
  
  for (const header of setCookieHeaders) {
    const match = header.match(/^([^=]+)=([^;]*)/);
    if (match) {
      const [, name, value] = match;
      if (name.trim() === 'accessToken' || name.trim() === 'refreshToken') {
        const trimmedName = name.trim();
        if (value && value.length > 0) {
          cookies[trimmedName] = value;
        } else {
          cookies[trimmedName] = null;
        }
      }
    }
  }
}

/**
 * Build cookie header string from cookie jar
 */
function buildCookieHeader() {
  const parts = [];
  if (cookies.accessToken) parts.push(`accessToken=${cookies.accessToken}`);
  if (cookies.refreshToken) parts.push(`refreshToken=${cookies.refreshToken}`);
  return parts.join('; ');
}

/**
 * Make HTTP request simulating browser behavior
 */
async function request(method, path, body = null, options = {}) {
  const url = `${BASE_URL}${path}`;
  
  const headers = {
    'Content-Type': 'application/json',
    'Accept': 'application/json',
    'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7) TBV-Endpoint-Tester/1.0',
    ...options.headers
  };
  
  const cookieHeader = buildCookieHeader();
  if (cookieHeader) {
    headers['Cookie'] = cookieHeader;
  }
  
  const fetchOptions = {
    method,
    headers,
    credentials: 'include'
  };
  
  if (body && method !== 'GET') {
    fetchOptions.body = JSON.stringify(body);
  }
  
  const response = await fetch(url, fetchOptions);
  parseCookies(response);
  
  let data = null;
  const contentType = response.headers.get('content-type');
  if (contentType?.includes('application/json')) {
    try {
      data = await response.json();
    } catch (e) {
      data = null;
    }
  }
  
  return { status: response.status, data, ok: response.ok };
}

/**
 * Refresh access token using refresh token
 */
async function refreshToken() {
  if (!cookies.refreshToken) {
    console.log('  ⚠️  No refresh token available');
    return false;
  }
  
  console.log('  🔄 Refreshing access token...');
  const res = await request('POST', '/api/auth/refresh');
  
  if (res.ok) {
    console.log('  ✅ Token refreshed successfully');
    return true;
  }
  
  console.log('  ❌ Token refresh failed');
  return false;
}

/**
 * Make authenticated request with auto-refresh on 401
 */
async function authRequest(method, path, body = null, options = {}) {
  let res = await request(method, path, body, options);
  
  if (res.status === 401 && cookies.refreshToken) {
    const refreshed = await refreshToken();
    if (refreshed) {
      res = await request(method, path, body, options);
    }
  }
  
  return res;
}

/**
 * Escape CSV value
 */
function csvEscape(value) {
  if (value === null || value === undefined) return '';
  const str = String(value);
  if (str.includes(',') || str.includes('"') || str.includes('\n')) {
    return `"${str.replace(/"/g, '""')}"`;
  }
  return str;
}

/**
 * Convert array of objects to CSV string
 */
function toCSV(items, maxItems = 10) {
  if (!items || items.length === 0) return '     (no items)';
  
  const displayItems = items.slice(0, maxItems);
  const allKeys = new Set();
  
  // Collect all keys from all items
  for (const item of displayItems) {
    if (typeof item === 'object' && item !== null) {
      Object.keys(item).forEach(k => allKeys.add(k));
    }
  }
  
  const headers = Array.from(allKeys);
  if (headers.length === 0) return '     (no displayable data)';
  
  const lines = [];
  lines.push('     ' + headers.map(csvEscape).join(','));
  
  for (const item of displayItems) {
    const values = headers.map(h => {
      const v = item[h];
      if (typeof v === 'object' && v !== null) {
        return csvEscape(JSON.stringify(v));
      }
      return csvEscape(v);
    });
    lines.push('     ' + values.join(','));
  }
  
  if (items.length > maxItems) {
    lines.push(`     ... (${items.length - maxItems} more items)`);
  }
  
  return lines.join('\n');
}

/**
 * Log response data - verbose mode logs CSV
 */
function logResponse(data, options = {}) {
  if (!data) {
    if (VERBOSE) console.log('     Response: null');
    return;
  }
  
  // Extract pagination metadata
  const meta = {};
  if (data.pagination) {
    meta.page = data.pagination.page || data.pagination.currentPage || 1;
    meta.totalPages = data.pagination.totalPages || '?';
    meta.total = data.pagination.total;
    meta.limit = data.pagination.limit || data.pagination.pageSize;
    meta.hasNext = data.pagination.hasNext;
    meta.hasPrev = data.pagination.hasPrev;
  } else if (data.total !== undefined) {
    meta.total = data.total;
  } else if (data.totalEvents !== undefined) {
    meta.totalEvents = data.totalEvents;
  }
  
  // Find the items array
  const itemArrayKeys = ['profiles', 'events', 'folders', 'leadLists', 'companies', 'filters', 
                         'users', 'sessions', 'templates', 'tags', 'quotas', 'contacts'];
  let itemsKey = null;
  let items = null;
  
  for (const key of itemArrayKeys) {
    if (data[key] && Array.isArray(data[key])) {
      itemsKey = key;
      items = data[key];
      break;
    }
  }
  
  // Check nested data.tags
  if (!items && data.data?.tags && Array.isArray(data.data.tags)) {
    itemsKey = 'tags';
    items = data.data.tags;
  }
  
  // Direct array response
  if (!items && Array.isArray(data)) {
    itemsKey = 'items';
    items = data;
  }
  
  // Build metrics string
  const metrics = [];
  if (meta.page) metrics.push(`Page: ${meta.page}/${meta.totalPages}`);
  if (meta.total !== undefined) metrics.push(`Total: ${meta.total}`);
  if (meta.totalEvents !== undefined) metrics.push(`Total Events: ${meta.totalEvents}`);
  if (items) metrics.push(`${itemsKey}: ${items.length} items`);
  
  if (metrics.length > 0) {
    console.log(`     📊 ${metrics.join(' | ')}`);
  }
  
  // Verbose mode: log CSV
  if (VERBOSE) {
    // Log metadata as CSV
    if (Object.keys(meta).length > 0) {
      console.log('\n     --- Pagination Metadata ---');
      console.log('     ' + Object.keys(meta).map(csvEscape).join(','));
      console.log('     ' + Object.values(meta).map(csvEscape).join(','));
    }
    
    // Log items as CSV
    if (items && items.length > 0) {
      console.log(`\n     --- ${itemsKey} (first page, max 10) ---`);
      console.log(toCSV(items, 10));
    }
    
    // For non-paged single object responses
    if (!items && typeof data === 'object' && !Array.isArray(data) && Object.keys(meta).length === 0) {
      // It's a single object response (like profile detail, stats, etc.)
      const keys = Object.keys(data).filter(k => !itemArrayKeys.includes(k));
      if (keys.length > 0 && keys.length <= 20) {
        console.log('\n     --- Response Data ---');
        console.log('     ' + keys.map(csvEscape).join(','));
        console.log('     ' + keys.map(k => {
          const v = data[k];
          if (typeof v === 'object' && v !== null) {
            return csvEscape(JSON.stringify(v));
          }
          return csvEscape(v);
        }).join(','));
      }
    }
  }
}

/**
 * Test an endpoint and record result
 */
async function testEndpoint(name, method, path, body = null, expectedStatus = [200, 201], options = {}) {
  const expected = Array.isArray(expectedStatus) ? expectedStatus : [expectedStatus];
  
  try {
    const res = await authRequest(method, path, body);
    const statusOk = expected.includes(res.status);
    const isUnauthorized = res.status === 401 || res.status === 403;
    
    if (statusOk) {
      console.log(`  ✅ ${name}`);
      console.log(`     ${method} ${path} → ${res.status}`);
      logResponse(res.data);
      res.passed.push({ name, method, path, status: res.status });
      return res;
    } else if (isUnauthorized) {
      console.log(`  ❌ ${name} - UNAUTHORIZED`);
      console.log(`     ${method} ${path} → ${res.status}`);
      if (res.data?.error) {
        console.log(`     Error: ${res.data.error}`);
      }
      res.failed.push({ name, method, path, status: res.status, error: 'Unauthorized' });
      return res;
    } else {
      console.log(`  ⚠️  ${name}`);
      console.log(`     ${method} ${path} → ${res.status} (expected ${expected.join('/')})`);
      if (res.data?.error) {
        console.log(`     Error: ${res.data.error}`);
      }
      if (VERBOSE && res.data) {
        console.log(`     Response: ${JSON.stringify(res.data)}`);
      }
      res.skipped.push({ name, method, path, status: res.status, note: `Expected ${expected.join('/')}` });
      return res;
    }
  } catch (error) {
    console.log(`  ❌ ${name} - ERROR`);
    console.log(`     ${method} ${path}`);
    console.log(`     ${error.message}`);
    res.failed.push({ name, method, path, error: error.message });
    return null;
  }
}

main().catch(error => {
  console.error('Fatal error:', error);
  process.exit(1);
});
